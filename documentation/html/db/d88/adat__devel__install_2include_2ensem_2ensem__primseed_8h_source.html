303 template<class T1, class T2>
304 inline typename BinaryReturn<PSeed<T1>, PSeed<T2>, OpMultiply>::Type_t
305 operator*(const PSeed<T1>& s1, const PSeed<T2>& s2)
306 {
307  typename BinaryReturn<PSeed<T1>, PSeed<T2>, OpMultiply>::Type_t d;
308  typedef typename BinaryReturn<T1, T2, OpMultiply>::Type_t T;
309  typedef typename InternalScalar<T>::Type_t S;
310  T i0, i1, i2, i3;
311 
312  /* i3 = s1(3)*s2(0) + s1(2)*s2(1) + s1(1)*s2(2) + s1(0)*s2(3) */
313  i3 = s1.elem(3) * s2.elem(0);
314  i3 += s1.elem(2) * s2.elem(1);
315  i3 += s1.elem(1) * s2.elem(2);
316  i3 += s1.elem(0) * s2.elem(3);
317 
318  /* i2 = s1(2)*s2(0) + s1(1)*s2(1) + s1(0)*s2(2) */
319  i2 = s1.elem(2) * s2.elem(0);
320  i2 += s1.elem(1) * s2.elem(1);
321  i2 += s1.elem(0) * s2.elem(2);
322 
323  /* i1 = s1(1)*s2(0) + s1(0)*s2(1) */
324  i1 = s1.elem(1) * s2.elem(0);
325  i1 += s1.elem(0) * s2.elem(1);
326 
327  /* i0 = s1(0)*s2(0) */
328  i0 = s1.elem(0) * s2.elem(0);
329 
330  /* dest(0) = mod(i0, 4096) */
331  d.elem(0) = i0 & S(4095);
332 
333  /* i1 = i1 + i0/4096 */
334  i1 += i0 >> S(12);
335 
336  /* dest(1) = mod(i1, 4096) */
337  d.elem(1) = i1 & S(4095);
338 
339  /* i2 = i2 + i1/4096 */
340  i2 += i1 >> S(12);
341 
342  /* dest(2) = mod(i2, 4096) */
343  d.elem(2) = i2 & S(4095);
344  /* i3 = i3 + i2/4096 */
345  i3 += i2 >> S(12);
346 
347  /* dest(3) = mod(i3, 2048) */
348  d.elem(3) = i3 & S(2047);
349 
350  return d;
351 }

1249 contract(const OTensor<T>& s1, const Array<int>& nn)
1250 {

1252  if (s1.size().size() == 0 || nn.size() == 0)
1253  {
1254  std::cerr << __func__ << ": Invalid OTensor size" << std::endl;
1255  exit(1);
1256  }
1257  int nd = s1.size().size() - nn.size();
1258  if (nd <= 0)
1259  {
1260  std::cerr << __func__ << ": Invalid OTensor contract" << std::endl;
1261  exit(1);
1262  }
1263 
1264  // Check indices
1265  for(int i=0; i < nn.size(); ++i)
1266  {
1267  if (nn[i] >= s1.size().size())
1268  {
1269  std::cerr << "Contract index out of bounds" << std::endl;
1270  exit(1);
1271  }
1272  if (s1.size()[nn[i]] != s1.size()[nn[0]])
1273  {
1274  std::cerr << "Contract indices not equal lengths" << std::endl;
1275  exit(1);
1276  }
1277  }
1278 
1279  Array<int> free_ind(nd); // complement of mask
1280 
1281  // New array size
1282  Array<int> nz(nd);
1283  for(int i=0,j=0,k=0; i < s1.size().size(); ++i)
1284  {
1285  if (nn[j] == i)
1286  {
1287  j++;
1288  }
1289  else
1290  {
1291  free_ind[k] = i;
1292  nz[k++] = s1.size()[i];
1293  }
1294  }
1295 
1296  // Resize
1297  d.resize(nz);
1298 
1299  // Reset
1300  for(int i=0; i < d.numElem(); ++i)
1301  zero_rep(d.elem(i));
1302 
1303  // Do the contraction
1304  Array<int> ind(s1.size().size());
1305 
1306  for(int ipos0=0; ipos0 < s1.numElem(); ++ipos0)
1307  {
1308  bool diag = true;
1309  {
1310  // Decompose the indices of the source
1311  int ipos = ipos0;
1312  for(int i=0; i < s1.size().size(); ++i)
1313  {
1314  ind[i] = ipos % s1.size()[i];
1315  ipos /= s1.size()[i];
1316  }
1317 
1318  // Check if the summed indices are in fact the same
1319  for(int i=0; i < nn.size(); ++i)
1320  {
1321  if (ind[nn[i]] != ind[nn[0]])
1322  {
1323  diag = false;
1324  break;
1325  }
1326  }
1327  }
1328 
1329  // If this is a diagonal term, add it onto the result
1330  if (diag)
1331  {
1332  // Find the location inside the target
1333  int order = 0;
1334  for(int i=nz.size()-1; i >= 1; --i)
1335  order = nz[i-1]*(ind[free_ind[i]] + order);
1336 
1337  order += ind[free_ind[0]];
1338 
1339  d.elem(order) += s1.elem(ipos0);
1340  }
1341  }
1342 
1343  return d;
1344 }

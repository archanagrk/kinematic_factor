238 template<class T1, class T2, class T3>
239 inline typename TrinaryReturn<PColorVector<T1,3>, PColorVector<T2,3>, PColorVector<T3,3>, FnColorContract>::Type_t
240 colorContract(const PColorVector<T1,3>& s1, const PColorVector<T2,3>& s2, const PColorVector<T3,3>& s3)
241 {
242  typename TrinaryReturn<PColorVector<T1,3>, PColorVector<T2,3>, PColorVector<T3,3>, FnColorContract>::Type_t d;
243 
244  // Permutations: +(0,1,2)+(1,2,0)+(2,0,1)-(1,0,2)-(0,2,1)-(2,1,0)
245 
246  // d = \epsilon^{i,j,k} V1^{i} V2^{j} V3^{k}
247  d.elem() = (s1.elem(0)*s2.elem(1)
248  - s1.elem(1)*s2.elem(0))*s3.elem(2)
249  + (s1.elem(1)*s2.elem(2)
250  - s1.elem(2)*s2.elem(1))*s3.elem(0)
251  + (s1.elem(2)*s2.elem(0)
252  - s1.elem(0)*s2.elem(2))*s3.elem(1);
253 
254  return d;
255 }

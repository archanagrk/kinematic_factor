221  * Internal byte swapping code if we are using little endian
222  */
223 /*
224  * Little endian <==> big endian 32-bit swap macros.
225  * M_32_SWAP swap a memory location
226  * P_32_SWAP swap a referenced memory location
227  * P_32_COPY swap from one location to another
228  */
229 #define M_32_SWAP(a) { \
230  unsigned int _tmp = a; \
231  ((char *)&a)[0] = ((char *)&_tmp)[3]; \
232  ((char *)&a)[1] = ((char *)&_tmp)[2]; \
233  ((char *)&a)[2] = ((char *)&_tmp)[1]; \
234  ((char *)&a)[3] = ((char *)&_tmp)[0]; \
235 }
236 #define P_32_SWAP(a) { \
237  unsigned int_tmp = *(unsigned int *)a; \
238  ((char *)a)[0] = ((char *)&_tmp)[3]; \
239  ((char *)a)[1] = ((char *)&_tmp)[2]; \
240  ((char *)a)[2] = ((char *)&_tmp)[1]; \
241  ((char *)a)[3] = ((char *)&_tmp)[0]; \
242 }
243 #define P_32_COPY(a, b) { \
244  ((char *)&(b))[0] = ((char *)&(a))[3]; \
245  ((char *)&(b))[1] = ((char *)&(a))[2]; \
246  ((char *)&(b))[2] = ((char *)&(a))[1]; \
247  ((char *)&(b))[3] = ((char *)&(a))[0]; \
248 }

252  * M_16_SWAP swap a memory location

255  */
256 #define M_16_SWAP(a) { \
